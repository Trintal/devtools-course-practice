#############################################
##### Testing
#############################################

set(prefix "${MODULE}")

add_test(
    NAME ${prefix}_can_Run
    COMMAND ${APPLICATION}
)
set_tests_properties (${prefix}_can_Run PROPERTIES
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_can_Print_Help_Without_Arguments
    COMMAND ${APPLICATION}
)
set_tests_properties (${prefix}_can_Print_Help_Without_Arguments PROPERTIES
    PASS_REGULAR_EXPRESSION "This is a calculation of area of geometric figures application"
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_can_Check_Number_Of_Arguments
    COMMAND ${APPLICATION} Cone 2 3
)
set_tests_properties (${prefix}_can_Check_Number_Of_Arguments PROPERTIES
    PASS_REGULAR_EXPRESSION "ERROR: Should be not more than 3"
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_can_Calculate_Cone_Area
    COMMAND ${APPLICATION} Cone 2 3
)
set_tests_properties (${prefix}_can_Calculate_Cone_Area PROPERTIES
    PASS_REGULAR_EXPRESSION "Area of Cone: 31,41592653589793238463"
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Cant_Calculate_Cone_Negative
    COMMAND ${APPLICATION} Cone -2 3
)
set_tests_properties (${prefix}_Cant_Calculate_Cone_Negative PROPERTIES
    PASS_REGULAR_EXPRESSION "Number can't be less then zero"
    LABELS "${MODULE}")
